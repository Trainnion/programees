const project_loader = async (File_directory) => {

    let existing_parent_code_Contaner = document.querySelector(".parent_code_container");
    if (existing_parent_code_Contaner) {
        existing_parent_code_Contaner.remove();
    }

    let information_about_member = document.getElementById("information_about_member");
    let parent_code_Contaner = document.createElement("div");
    parent_code_Contaner.className = "parent_code_container";
    information_about_member.appendChild(parent_code_Contaner);

    parent_code_Contaner.innerHTML = `
        <h2>Project Code</h2>
        <pre id="code-container" class="code-container"></pre>
    `;

    try {
        const response = await fetch(File_directory);
        if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);


        const code = await response.text();
        document.getElementById('code-container').textContent = code;
    } catch (err) {
        // Handle errors, such as file not found or network issues
      document.getElementById('code-container').textContent = 'Error loading code: ' + err.message;
      console.error(err);
      return
    };
}

export default project_loader;
